# يحتوي هذا الملف على رسائل الخطأ التي تُعرض للمستخدم عند استخدام تسجيل الدخول عبر HTTP/HTTPS.
# يمكن تغيير هذه الرسائل لجعل واجهة المستخدم أكثر وُدية، بما في ذلك الترجمات إلى لغات مختلفة.
#
# يمكن استخدام متغيرات مختلفة هنا أيضًا. الأكثر استخدامًا هي:
#   $(error-orig)   - رسالة الخطأ الأصلية من نقطة الاتصال
#   $(ip)           - عنوان IP للعميل
#   $(username)     - اسم المستخدم للعميل الذي يحاول تسجيل الدخول

# internal-error
# لا ينبغي أن يحدث أبداً. إذا حدث ذلك، سيتم عرض صفحة خطأ
# تعرض هذه الرسالة (error-orig ستصف ما حدث)

internal-error = خطأ داخلي ($(error-orig))

# config-error
# لا ينبغي أن يحدث إذا تم تكوين نقطة الاتصال بشكل صحيح.

config-error = خطأ في التكوين ($(error-orig))

# not-logged-in
# سيحدث ذلك إذا طلب المستخدم صفحة الحالة أو تسجيل الخروج،
# وهو في الواقع ليس مسجلاً للدخول

not-logged-in = أنت لست مسجلاً للدخول (ip $(ip))

# ippool-empty
# يتم تعيين عنوان IP للمستخدم من مجموعة عناوين IP، ولكن لا توجد
# عناوين متاحة في هذه المجموعة

ippool-empty = لا يمكن تعيين عنوان IP - لا توجد عناوين متاحة في المجموعة

# shutting-down
# عند تنفيذ إيقاف التشغيل، لا يتم قبول عملاء جدد

shutting-down = خدمة نقطة الاتصال قيد الإيقاف

# user-session-limit
# إذا كان للملف الشخصي للمستخدم حد لمشاركة المستخدمين، فسيتم عرض هذا الخطأ
# بعد الوصول إلى هذا الحد

user-session-limit = لا مزيد من الجلسات مسموحة للمستخدم $(username)

# license-session-limit
# بناءً على الترخيص، يتم تحديد عدد العملاء النشطين في نقطة الاتصال
# بكمية معينة. إذا تم الوصول إلى هذا الحد، يتم عرض الخطأ التالي.

license-session-limit = تم الوصول إلى حد الجلسات ($(error-orig))

# wrong-mac-username
# إذا كان اسم المستخدم يشبه عنوان MAC (12:34:56:78:9a:bc)، ولكنه ليس
# عنوان MAC الخاص بهذا العميل، يتم رفض تسجيل الدخول

wrong-mac-username = اسم المستخدم غير صالح ($(username)): هذا العنوان MAC ليس لك

# chap-missing
# إذا تم استخدام طريقة تسجيل الدخول HTTP-CHAP، ولكن برنامج نقطة الاتصال لم يتلقَ
# كلمة المرور المشفرة، يتم عرض رسالة الخطأ هذه.
# الأسباب المحتملة للفشل:
#   - JavaScript غير مفعل في متصفح الويب؛
#   - صفحة login.html غير صالحة؛
#   - قيمة التحدي انتهت صلاحيتها على الخادم (أكثر من ساعة من عدم النشاط)؛
#   - تم إزالة طريقة تسجيل الدخول HTTP-CHAP مؤخراً؛
# إذا كان JavaScript مفعلاً وكانت صفحة login.html صالحة،
# فعادةً ما تحل إعادة محاولة تسجيل الدخول هذه المشكلة.

chap-missing = متصفح الويب لم يرسل استجابة التحدي (حاول مرة أخرى، فعّل JavaScript)

# invalid-username
# الحالة الأكثر شيوعاً لاسم المستخدم أو كلمة المرور غير الصالحة. إذا أرسل
# خادم RADIUS سلسلة خطأ مع رسالة Access-Reject، فسيتم
# تجاوز هذا الإعداد.

invalid-username = اسم المستخدم أو كلمة المرور غير صالحين

# invalid-mac
# يمكن ربط المستخدمين المحليين (على خادم نقطة الاتصال) بعنوان MAC معين. إذا تم
# محاولة تسجيل الدخول من عنوان MAC مختلف، يتم عرض رسالة الخطأ هذه.

invalid-mac = المستخدم $(username) غير مسموح له بتسجيل الدخول من هذا العنوان MAC

# uptime-limit, traffic-limit
# للمستخدمين المحليين في نقطة الاتصال في حالة الوصول إلى الحدود

uptime-limit = المستخدم $(username) وصل إلى حد زمن التشغيل
traffic-limit = المستخدم $(username) وصل إلى حد حركة المرور

# radius-timeout
# يتم التحقق من المستخدم بواسطة خادم RADIUS، ولكن لم يتم استلام أي استجابة منه،
# يتم عرض الخطأ التالي.

radius-timeout = خادم RADIUS لا يستجيب

# auth-in-progress
# التوثيق قيد التنفيذ. العميل قد أصدر بالفعل طلب توثيق
# لم يكتمل بعد.

auth-in-progress = التوثيق قيد التنفيذ، حاول مرة أخرى لاحقاً

# radius-reply
# خادم RADIUS أعاد رسالة خطأ مخصصة

radius-reply = $(error-orig)
